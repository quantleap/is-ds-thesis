create schema dsc;

create view dsc.company_insolvents as
	select id, 
    	   case_number, 
           start_date_insolvency, 
           end_date_insolvency, 
           person_family_name as business_name, 
           treating_court_code, 
           treating_subsidiary_code, 
           supervisory_judge,
           person_chamber_of_commerce_registration_number as coc_code
    from insolvents
   	where person_legal_personality = 'rechtspersoon';
